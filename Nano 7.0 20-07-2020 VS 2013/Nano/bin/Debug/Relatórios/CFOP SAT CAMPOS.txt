select  Nfe.CFOP, convert(numeric(15,2),sum(Pedido.Qtd * Convert(Numeric(15,2),Pedido.ValorUnitario))) as Valor, convert(varchar(10),SatFiscal.Data,103) as data FROM (SatFiscal left join Pedido on Satfiscal.CodOrdem = Pedido.CodOrdem) left join Nfe on Pedido.CodProd = Nfe.CodProd where SATFISCAL.Data >= 'DATAINICIALRELATORIO' AND SATFISCAL.Data <= 'DATAFINALRELATORIO 23:59:00' and Nfe.regime = 'SIMPLES NACIONAL' group by  Nfe.CFOP, convert(varchar(10),SatFiscal.Data,103) order by data, cfop desc
