SET DATEFORMAT dmy (SELECT convert(integer,Recebimento.CodOrdem) as 'Nº Venda', Pagamento, 
Convert(Numeric(15,2),Recebimento.Valor) as 'Valor Vendido', Convert(Numeric(15,2),Recebimento.Valor) as 'Valor Líquido', 
Recebimento.Data, Recebimento.CodConfig as 'Cód. Loja', convert(bit,'True') as Recebido, Recebimento.Data as 'Data Venda',
Cliente.Nome as 'Cliente', Cliente.CPF as 'CPF/CNPJ', Cliente.Email 
FROM ((Recebimento left join Pagamento on Recebimento.Pagamento = Pagamento.Descricao) left join Total on 
Total.CodOrdem = recebimento.CodOrdem and Total.CodConfig = Recebimento.CodConfig) left join Cliente on Total.CodCli = Cliente.Codigo where 
Recebimento.FormaPagamento = 'A VISTA' and Prazo = 'False' and Recebimento.Data >= 'DATAINICIALRELATORIO' AND 
Recebimento.Data <= 'DATAFINALRELATORIO')
UNION ALL (SELECT convert(integer,ContaReceber.CodOrdem) as 'Nº Venda', Pagamento, 
Convert(Numeric(15,2),ValorParcela) as 'Valor Vendido', Convert(Numeric(15,2),ValorParcela) as 'Valor Líquido',
 Vencimento as Data, ContaReceber.CodConfig as 'Cód. Loja', Recebido, Total.Data as 'Data Venda',
Cliente.Nome as 'Cliente', Cliente.CPF as 'CPF/CNPJ', Cliente.Email 
FROM (ContaReceber left join Total on ContaReceber.CodOrdem = Total.CodOrdem and ContaReceber.CodConfig = Total.CodConfig) left join Cliente on ContaReceber.CodCli = Cliente.Codigo 
where RecebidoOK = 'False' and Total.Data >= 'DATAINICIALRELATORIO' AND Total.Data <= 'DATAFINALRELATORIO')
UNION ALL (SELECT convert(integer,ContaReceber.CodOrdem) as 'Nº Venda', Pagamento, 
Convert(Numeric(15,2),ValorParcela) as 'Valor Vendido', Convert(Numeric(15,2),ValorRecebido) as 'Valor Líquido', 
Recebimento as Data, ContaReceber.CodConfig as 'Cód. Loja', Recebido, Total.Data as 'Data Venda',
Cliente.Nome as 'Cliente', Cliente.CPF as 'CPF/CNPJ', Cliente.Email 
FROM (ContaReceber left join Total on ContaReceber.CodOrdem = Total.CodOrdem and ContaReceber.CodConfig = Total.CodConfig)  left join Cliente on ContaReceber.CodCli = Cliente.Codigo 
where RecebidoOK = 'True' and Total.Data >= 'DATAINICIALRELATORIO' AND Total.Data <= 'DATAFINALRELATORIO')
UNION ALL (SELECT convert(integer,Futuro.CodOrdem) as 'Nº Venda', Pagamento, 
Convert(Numeric(15,2),Futuro.Total) as 'Valor Vendido', Convert(Numeric(15,2),Futuro.TotalDesconto) as 'Valor Líquido', 
Futuro.Data, Futuro.CodConfig as 'Cód. Loja', Futuro.Recebido as Recebido, Total.Data as 'Data Venda',
Cliente.Nome as 'Cliente', Cliente.CPF as 'CPF/CNPJ', Cliente.Email 
FROM (Futuro left join Total on Futuro.CodOrdem = Total.CodOrdem and Futuro.CodConfig = Total.CodConfig) left join Cliente on Total.CodCli = Cliente.Codigo where 
Total.Data >= 'DATAINICIALRELATORIO' AND Total.Data <= 'DATAFINALRELATORIO')
