Select ItensConsig.CodOrdens as CodOrdem, TotalConsig.Data, ItensConsig.Qtd, ItensConsig.Produto, Convert(Numeric(15,3),ValorUnitario) as ValorUnitario, Convert(Numeric(15,3),UniSemDesc) as UniSemDesc, Convert(Numeric(15,2), ValorTotal) as ValorTotal, Cliente.Nome + ' Endereço: ' + cliente.endereco + ', ' + cliente.Numero + ' - ' + Cliente.Bairro + ' ' + cliente.CEP + ' ' + cliente.Cidade as Cliente, TotalConsig.Status, TotalConsig.CodFunc as Vendedor, ItensConsig.Custo as ValorCusto from (((ItensConsig left join Produto on ItensConsig.CodigoProduto = Produto.Codigo) left join TotalConsig on ItensConsig.CodOrdens = TotalConsig.CodOrdens and ItensConsig.CodConfig = TotalConsig.CodConfig) left Join Cliente On TotalConsig.CodCli = Cliente.Codigo) left join Total on TotalConsig.CodOrdens = Total.CodPedido
